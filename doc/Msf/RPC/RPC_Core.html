def rpc_thread_list
  res = {}
  framework.threads.each_index do |i|
    t = framework.threads[i]
    next if not t
    res[i] = {
      :status   => (t.status || "dead"),
      :critical => t[:tm_crit] ? true : false,
      :name     => t[:tm_name].to_s,
      :started  => t[:tm_time].to_s
    }
  end
  res
end
