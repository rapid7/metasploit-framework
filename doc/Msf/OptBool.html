def valid?(value)
  return false if empty_required_value?(value)

  if ((value != nil and
      (value.to_s.empty? == false) and
      (value.to_s.match(/^(y|yes|n|no|t|f|0|1|true|false)$/i) == nil)))
    return false
  end

  true
end
