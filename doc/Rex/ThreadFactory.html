def self.spawn(name, crit, *args, &block)
  if @@provider
    if block
      return @@provider.spawn(name, crit, *args){ |*args_copy| block.call(*args_copy) }
    else
      return @@provider.spawn(name, crit, *args)
    end
  else
    t = nil
    if block
      t = ::Thread.new(*args){ |*args_copy| block.call(*args_copy) }
    else
      t = ::Thread.new(*args)
    end
    t[:tm_name] = name
    t[:tm_crit] = crit
    t[:tm_time] = Time.now
    t[:tm_call] = caller
    return t
  end

end
