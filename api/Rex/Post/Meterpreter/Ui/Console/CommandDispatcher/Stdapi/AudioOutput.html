def cmd_play(*args)
  if args.length == 0
    print_line('Please specify a path to an audio file')
    return
  end

  audio_path = args[0]

  print_status("Playing #{audio_path}...")
  client.audio_output.play_file(audio_path)
  print_status('Done')
end
