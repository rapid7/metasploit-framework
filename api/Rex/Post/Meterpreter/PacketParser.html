def recv(sock)
  raw = nil
  if self.packet.raw_bytes_required > 0
    while (raw = sock.read(self.packet.raw_bytes_required))
      self.packet.add_raw(raw)
      break if self.packet.raw_bytes_required == 0
    end
  end

  if self.packet.raw_bytes_required > 0
    if raw == nil
      raise EOFError
    else
      return nil
    end
  end

  packet = self.packet
  reset
  packet
end
