def espia_image_get_dev_screen
  request = Packet.create_request(COMMAND_ID_ESPIA_IMAGE_GET_DEV_SCREEN)
  response = client.send_request( request )
  if response.result == 0
    response.get_tlv_value(TLV_TYPE_DEV_SCREEN)
  else
    nil
  end
end
