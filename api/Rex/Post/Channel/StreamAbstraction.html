def read(length = nil)
  if closed?
    raise IOError, 'Channel has been closed.', caller
  end

  buf = ''
  length = 65536 if length.nil?

  begin
    buf << lsock.recv(length - buf.length) while buf.length < length
  rescue StandardError
    buf = nil if buf.empty?
  end

  buf
end
