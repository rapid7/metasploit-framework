def pack
  path = @entry.relative_path

  ret = [ SIGNATURE, ZIP_VERSION ].pack('Vv')
  ret << @entry.flags.pack
  ret << @entry.info.pack
  ret << [ path.length, @entry.xtra.length ].pack('vv')
  ret << path
  ret << @entry.xtra
  ret
end
