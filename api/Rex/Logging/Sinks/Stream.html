def log(sev, src, level, msg) # :nodoc:
  if sev == LOG_RAW
    stream.write(msg)
  else
    stream.write("[#{get_current_timestamp}] [#{log_code_for(sev)}(#{level})] #{src}: #{msg}\n")
  end

  stream.flush
end

def log_code_for(sev)
  code = 'i'

  case sev
  when LOG_DEBUG
    code = 'd'
  when LOG_ERROR
    code = 'e'
  when LOG_INFO
    code = 'i'
  when LOG_WARN
    code = 'w'
  end

  code
end
