def self.bytes_to_int(bytes)
  bytes.each_byte.reduce(0) { |acc, byte| (acc << 8) | byte }
end

def self.int_to_bytes(num)
  bytes = []

  while num > 0
    bytes.unshift(num & 0xff)
    num >>= 8
  end

  bytes.pack("C*")
end
