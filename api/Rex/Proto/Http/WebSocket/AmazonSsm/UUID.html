def self.pack(sbuf)
  parts = sbuf.split('-').map do |seg|
    seg.chars.each_slice(2).map {|e| "\\x#{e.join}"}.join
  end
  [3, 4, 0, 1, 2].map do |part|
    Rex::Text.hex_to_raw(parts[part])
  end.join
end

def self.unpack(bbuf)
  sbuf = ""
  [8...12].each do |idx|
    sbuf << Rex::Text.to_hex(bbuf[idx])
  end
  sbuf << '-'
  [12...14].each do |idx|
    sbuf << Rex::Text.to_hex(bbuf[idx])
  end
  sbuf << '-'
  [14...16].each do |idx|
    sbuf << Rex::Text.to_hex(bbuf[idx])
  end
  sbuf << '-'
  [0...2].each do |idx|
    sbuf << Rex::Text.to_hex(bbuf[idx])
  end
  sbuf << '-'
  [2...8].each do |idx|
    sbuf << Rex::Text.to_hex(bbuf[idx])
  end
  sbuf.gsub("\\x",'')
end
