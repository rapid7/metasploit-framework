def coerce
  content.map do |item|
    case item.data.data_type
    when 'A'.ord # field-array
      [item.name.to_s, item.data.data.coerce]
    when 'F'.ord # field-table
      [item.name.to_s, item.data.data.coerce]
    else
      [item.name.to_s, item.data.data.snapshot]
    end
  end.to_h
end
