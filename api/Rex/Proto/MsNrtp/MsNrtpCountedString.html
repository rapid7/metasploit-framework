def set(v)
  self.string_data = v.bytes
  if v.encoding == Encoding::UTF_16LE
    self.string_encoding = 0
  elsif v.encoding == Encoding::UTF_8
    self.string_encoding = 1
  else
    raise ::EncodingError, 'strings must be UTF-8 or UTF-16'
  end
end
