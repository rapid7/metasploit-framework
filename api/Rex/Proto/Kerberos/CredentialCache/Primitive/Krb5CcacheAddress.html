def get
  v = IPAddr.new_ntoh(super)
  if self.class::FAMILY != ::Socket::AF_UNSPEC && self.class::FAMILY != v.family
    raise IPAddr::AddressFamilyError, 'address family mismatch'
  end

  v
end

def set(v)
  if v.is_a?(IPAddr)
    if self.class::FAMILY != ::Socket::AF_UNSPEC && self.class::FAMILY != v.family
      raise IPAddr::AddressFamilyError, 'address family mismatch'
    end

    v = v.hton
  end

  super
end
