def generate_message
  missing_buffer_names = @ul_types.map do |ul_type|
    Rex::Proto::Kerberos::Pac::Krb5PacElementType.const_name(ul_type)
  end
  "Missing Info Buffer(s): #{missing_buffer_names.join(', ')}"
end
