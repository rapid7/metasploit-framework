def present(serialized_message)
  lines = []
  case serialized_message
  when Hash
    append_hash(lines, serialized_message, indent: 0)
  when Array
    append_array(lines, serialized_message, indent: 0)
  else
    lines << serialized_message.to_s
  end
  lines.join("\n")
end
