def define(&block)
  klass = Class.new(self) do
    def self.inherited(subclass)
      proto = @proto

      subclass.instance_eval do
        @proto = proto
      end
    end
  end

  klass.module_eval(&block)

  klass
end

def [](name)

  raise ArgumentError, "no such field: #{name}" unless a
