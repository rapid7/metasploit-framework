def unwrap_sasl(data)
  length = data[0,4].unpack('N')[0]
  if length != data.length + 4
    raise ArgumentError.new('Invalid SASL structure')
  end

  data[4,length]
end

def wrap_sasl(data)
  length = [data.length].pack('N')

  length + data
end
