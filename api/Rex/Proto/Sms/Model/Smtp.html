def initialize(opts={})
  self.address     = opts[:address]
  self.port        = opts[:port]        || 25
  self.username    = opts[:username]
  self.password    = opts[:password]
  self.helo_domain = opts[:helo_domain] || 'localhost'
  self.login_type  = opts[:login_type]  || :login
  self.from        = opts[:from]        || ''
end
