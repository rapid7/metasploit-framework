def read(str="")
  raise DRDA::Error, "Input isn't a String." if !str.kind_of? String
  self.clear
  i = 0
  while(i < str.size)
    this_ddm = BASIC_DDM.new.read(str[i,str.size])
    self << this_ddm
    i += this_ddm.to_s.size
  end
  return self
end
