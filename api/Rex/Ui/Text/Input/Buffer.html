def gets
  # Initialize the line buffer
  line = ''

  # Read data one byte at a time until we see a LF
  while (true)
    break if line.include?("\n")

    # Read another character of input
    char = @sock.rsock.getc

    # Append this character to the string
    line << char
  end

  return line
end
