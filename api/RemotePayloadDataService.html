def update_payload(opts)
  path = PAYLOAD_API_PATH
  if opts && opts[:id]
    id = opts.delete(:id)
    path = "#{PAYLOAD_API_PATH}/#{id}"
  end
  json_to_mdm_object(self.put_data(path, opts), PAYLOAD_MDM_CLASS)
end
