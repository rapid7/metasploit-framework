def <<( element )

  case element
    when Auxiliary::Web::Path
      @paths << element
    when Auxiliary::Web::Form
      @forms << element
    when Mdm::WebForm
      self.<< element.method.to_s.downcase == 'path' ?
        Auxiliary::Web::Path.from_model( element ) :
                Auxiliary::Web::Form.from_model( element )
  end
end

def update( options = {} )
  options.each { |k, v| send( "#{k}=", v ) }

  @forms ||= []
  @paths ||= []

  self
end
