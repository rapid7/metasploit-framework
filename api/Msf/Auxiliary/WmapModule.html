def distance(a, b)
  case
  when a.empty?
    b.length
  when b.empty?
    a.length
  else
    [(a[0] == b[0] ? 0 : 1) + distance(a[1..-1], b[1..-1]),
    1 + distance(a[1..-1], b),
    2 + distance(a, b[1..-1])].min
  end
end

def wmap_base_url
  res = (ssl ? "https://" : "http://")
  if datastore['VHOST'].nil?
    res << wmap_target_host
  else
    res << datastore['VHOST']
  end
  res << ":" + wmap_target_port.to_s
  res
end
