def create_request(host,port,ssl,meth,path,headers,query,body,respcode,resphead,response)
::ApplicationRecord.connection_pool.with_connection {
  req = ::Mdm::WmapRequest.create(
      :host => host,
      :address => host,
      :port => port,
      :ssl => ssl,
      :meth => meth,
      :path => path,
      :headers => headers,
      :query => query,
      :body => body,
      :respcode => respcode,
      :resphead => resphead,
      :response => response
    )
  #framework.events.on_db_request(rec)
}
end

}

  if selhost
    return selhost.host
  else
    return
  end
}
