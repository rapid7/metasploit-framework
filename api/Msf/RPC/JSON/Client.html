def method_missing(symbol, *args, **keyword_args, &block)
  # assemble method parameters
  if !args.empty? && !keyword_args.empty?
    params = args << keyword_args
  elsif !args.empty?
    params = args
  elsif !keyword_args.empty?
    params = keyword_args
  else
    params = nil
  end

  process_call_async(symbol, params)
end
