def self.parse(response, symbolize_names: true)
  id_key = symbolize_names ? :id : :id.to_s
  result_key = symbolize_names ? :result : :result.to_s

  id = response[id_key]
  result = response[result_key]

  Response.new(id: id, result: result, response: response)
end
