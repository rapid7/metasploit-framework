def to_h
  hash = {
      code: @code,
      message: @message
  }

  # process data member
  unless @data.nil?
    if @data.is_a?(String) || @data.kind_of?(Numeric) || @data.is_a?(Array) || @data.is_a?(Hash)
      hash[:data] = @data
    elsif @data.respond_to?(:to_h)
      hash[:data] = @data.to_h
    else
      hash[:data] = @data.to_s
    end
  end

  hash
end
