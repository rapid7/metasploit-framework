def initialize(info = {})
  super(
    update_info(
      info,
      'Compat' => {
        'Meterpreter' => {
          'Commands' => %w[
            stdapi_sys_config_sysinfo
          ]
        }
      }
    )
  )

  register_options( [
    Msf::OptInt.new('SESSION', [ true, 'The session to run this module on' ])
  ] , Msf::Post)
end
