def self.new(*args, &block)
  style, opts = args

  if style == Style::Default
    opts['Indent']  = 3
    if (!opts['Prefix'])
      opts['Prefix']  = "\n"
    end
    if (!opts['Postfix'])
      opts['Postfix'] = "\n"
    end
  end

  instance = super(opts, &block)
  if style == Style::Default
    instance.extend(DefaultStyle)
  end
  instance
end
