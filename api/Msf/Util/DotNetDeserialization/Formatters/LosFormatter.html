def self.generate(stream)
  stream = stream.to_binary_s
  formatted  = ObjectStateFormatter.new(token: TOKEN_BINARY_SERIALIZED).to_binary_s
  formatted << DotNetDeserialization.encode_7bit_int(stream.length)
  formatted << stream
end
