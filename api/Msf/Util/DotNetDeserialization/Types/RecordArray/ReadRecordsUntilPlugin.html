def do_read(io)
  loop do
    element = append_new_element
    begin
      element.do_read(io)
    rescue EOFError, IOError
      elements.pop
      break
    end

    break if elements[-1].record_type == Enums::RecordTypeEnum[:MessageEnd]
  end
end
