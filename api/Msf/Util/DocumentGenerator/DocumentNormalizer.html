def get_md_content(items, kb)
  @md_template ||= lambda {
    template = ''
    path = File.expand_path(File.join(Msf::Config.data_directory, 'markdown_doc', TEMPLATE_PATH))
    File.open(path, 'rb') { |f| template = f.read }
    return template
  }.call
  md_to_html(ERB.new(@md_template).result(binding()), kb)
end
