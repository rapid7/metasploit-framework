def prepends(buf)
  if datastore['PrependFork']
    buf = %Q^
      code = %(#{ Rex::Text.encode_base64(buf) }).unpack(%(m0)).first
      if RUBY_PLATFORM =~ /mswin|mingw|win32/
        inp = IO.popen(%(ruby), %(wb)) rescue nil
        if inp
          inp.write(code)
          inp.close
        end
      else
        if ! Process.fork()
          eval(code) rescue nil
        end
      end
    ^.strip.split("\n").map{|line| line.strip}.join("\n")
  end

  buf
end
