def compatible_encoders
  encoders = super()
  encoders2 = []

  encoders.each { |encname, encmod|
    if (!encname.include?('fnstenv_mov') && !encname.include?('shikata_ga_nai'))
      encoders2 << [ encname, encmod ]
    end
  }

  return encoders2;
end
