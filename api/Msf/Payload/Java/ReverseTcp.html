def stager_config(opts={})
  c = super
  ds = opts[:datastore] || datastore
  pass = ds["AESPassword"] || ''
  if pass != ""
    c << "AESPassword=#{pass}\n"
  end
  c << "LHOST=#{ds["LHOST"]}\n" if ds["LHOST"]
  c << "LPORT=#{ds["LPORT"]}\n" if ds["LPORT"]

  c
end
