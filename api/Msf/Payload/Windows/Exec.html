def command_string
  cmd = datastore['CMD'].to_s

  if cmd.empty? && cmd_required?
    fail_with(Msf::Module::Failure::BadConfig, 'CMD must not be empty')
  end

  cmd
end

def generate(_opts = {})
  raw = super
  raw = ''.dup if raw.nil?
  raw + command_string + "\x00"
end
