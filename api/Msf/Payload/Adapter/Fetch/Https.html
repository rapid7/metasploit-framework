def setup_handler
  unless datastore['FetchHandlerDisable']
    @fetch_service = start_https_fetch_handler(srvname)
    escaped_uri = ('/' + srvuri).gsub('//', '/')
    add_resource(@fetch_service, escaped_uri, @srvexe)
    unless @pipe_uri.nil?
      uri = ('/' + @pipe_uri).gsub('//', '/')
      add_resource(@fetch_service, uri, @pipe_cmd)
    end
  end
  super
end
