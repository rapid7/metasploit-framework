def unc
  path = "\\\\#{srvhost}"
  path << "\\#{srvuri.gsub('/', "\\").chomp("\\")}"
  path << "\\#{datastore['FETCH_FILENAME']}" if datastore['FETCH_FILENAME'].present?
  path
end
