def escape_arg(arg)
  quote_requiring = ['\\', '`', '(', ')', '<', '>', '&', '|', ' ', '@', '"', '$', ';']
  result = CommandShell._glue_cmdline_escape(arg, quote_requiring, "'", "\\'", "'")
  if result == ''
    result = "''"
  end

  result
end

def to_cmd(cmd_and_args)
  escaped = cmd_and_args.map { |arg| escape_arg(arg) }

  escaped.join(' ')
end
