def close
  cid = nil
  should_close_connection = false
  @mutex.synchronize do
    return if closed?

    cid = @cid
    should_close_connection = !@remote_closed
    @cid = nil
    @remote_closed = true
  end

  close_connection if should_close_connection
  stop_reader_thread
  @session.remove_channel(cid)
  cleanup_abstraction
end
