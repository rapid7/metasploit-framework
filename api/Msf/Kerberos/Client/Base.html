def build_client_name(opts = {})
  name = opts[:client_name] || ''
  name_type = opts[:client_type] || Rex::Proto::Kerberos::Model::NT_PRINCIPAL

  Rex::Proto::Kerberos::Model::PrincipalName.new(
    name_type: name_type,
    name_string: name.split('/')
  )
end
