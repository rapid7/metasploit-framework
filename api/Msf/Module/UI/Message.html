def print_prefix
  ret = ''
  if (datastore['TimestampOutput'] =~ /^(t|y|1)/i) || (
    framework && framework.datastore['TimestampOutput'] =~ /^(t|y|1)/i
  )
    prefix = "[#{Time.now.strftime("%Y.%m.%d-%H:%M:%S")}] "

    xn ||= datastore['ExploitNumber']
    xn ||= framework.datastore['ExploitNumber']
    if xn.is_a?(Fixnum)
      prefix << "[%04d] " % xn
    end

    ret = prefix
  end
  ret
end
