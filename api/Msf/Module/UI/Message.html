def print_prefix
  prefix = ''
  if datastore['TimestampOutput'] ||
      (framework && framework.datastore['TimestampOutput'])
    prefix << "[#{Time.now.strftime("%Y.%m.%d-%H:%M:%S")}] "

    xn ||= datastore['ExploitNumber']
    xn ||= framework.datastore['ExploitNumber']
    if xn.is_a?(Integer)
      prefix << "[%04d] " % xn
    end

  end
  prefix
end
