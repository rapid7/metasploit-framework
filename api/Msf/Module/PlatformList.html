Overview

This is a helper to an easy way to specify support platforms. It will take a list of strings or Msf::Module::Platform objects and build them into a list of Msf::Module::Platform objects. It also supports ranges based on relative ranks…

def initialize(*args)
  self.platforms = []

  args.each do |a|
    if a.is_a?(String)
      platforms << Msf::Module::Platform.find_platform(a)
    elsif a.is_a?(Range)
      a_begin = Msf::Module::Platform.find_platform(a.begin)
      a_end = Msf::Module::Platform.find_platform(a.end)
      range = (a_begin::Rank..a_end::Rank)

      a_begin.superclass.find_children.each do |c|
        platforms << c if range.include?(c::Rank)
      end
    else
      platforms << a
    end
  end
end

def &(other)
  l1 = platforms
  l2 = other.platforms
  total = l1.find_all { |m| l2.find { |mm| m <= mm } } |
          l2.find_all { |m| l1.find { |mm| m <= mm } }
  Msf::Module::PlatformList.from_a(total)
end

def supports?(platform_list)
  platform_list.platforms.each do |pl|
    supported = false
    platforms.each do |p|
      if p >= pl
        supported = true
        break
      end
    end
    return false unless supported
  end

  true
end
