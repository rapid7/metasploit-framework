def exec(method: :run, args: {}, &block)
  req = Message.new(method)
  req.params = args.dup

  b = Bridge.open(self.path, framework: self.framework).exec(req)

  if block
    begin
      while m = b.messages.pop
        block.call m
      end
    ensure
      b.close
    end
    return b.success?
  else
    return b
  end
end
