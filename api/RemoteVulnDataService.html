def update_vuln(opts)
  path = VULN_API_PATH
  if opts && opts[:id]
    id = opts.delete(:id)
    path = "#{VULN_API_PATH}/#{id}"
  end
  json_to_mdm_object(self.put_data(path, opts), VULN_MDM_CLASS)
end
