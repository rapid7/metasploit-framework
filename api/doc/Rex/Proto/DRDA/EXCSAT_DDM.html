def initialize(args={})
  self[:magic] = 0xd0
  self[:format] = 0x41
  self[:correlid] = 1
  self[:codepoint] = Constants::EXCSAT
  self[:mgrlvlls] = args[:mgrlvlls] || MGRLVLLS_PARAM.new.to_s
  self[:length] = (10 + self[:mgrlvlls].to_s.size)
  self[:length2] = self[:length]-6
end
