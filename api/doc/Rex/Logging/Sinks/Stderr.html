def log(sev, src, level, msg, from) # :nodoc:
  if (sev == LOG_RAW)
    $stderr.write(msg)
  else
    code = 'i'

    case sev
      when LOG_DEBUG
        code = 'd'
      when LOG_ERROR
        code = 'e'
      when LOG_INFO
        code = 'i'
      when LOG_WARN
        code = 'w'
    end
    $stderr.write("[#{get_current_timestamp}] [#{code}(#{level})] #{src}: #{msg}\n")
  end

  $stderr.flush
end
