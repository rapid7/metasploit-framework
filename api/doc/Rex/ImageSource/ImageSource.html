def read_asciiz(offset)
  # FIXME, make me better
  string = ''
  loop do
    begin
      char = read(offset, 1)
    rescue RangeError
      break
    end
    break if char.nil? || char == "\x00"
    offset += 1
    string << char
  end
  return string
end
