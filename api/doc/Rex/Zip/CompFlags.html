def initialize(gpbf, compmeth, timestamp)
  @gpbf = gpbf
  @compmeth = compmeth
  @mod_time = ((timestamp.hour << 11) | (timestamp.min << 5) | (timestamp.sec))
  @mod_date = (((timestamp.year - 1980) << 9) | (timestamp.mon << 5) | (timestamp.day))
end
