def scan_host(ip)
  if spoofed?
    datastore['ScannerRecvWindow'] = 0
    scanner_spoof_send(@probe, ip, datastore['RPORT'], datastore['SRCIP'], datastore['NUM_REQUESTS'])
  else
    scanner_send(@probe, ip, datastore['RPORT'])
  end
end
