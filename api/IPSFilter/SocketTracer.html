def ips_match(data)
  lp = localport
  rp = peerport

  SIGS.each do |s|
    begin
      r = Regexp.new(s[1])
      if (data.match(r))
        print_error "Matched IPS signature #{s[0]}"
        return true
      end
    rescue ::Exception => e
      print_error "Compiled error: #{s[1]}"
    end
  end

  return false
end

def read(length = nil, opts = {})
  r = super(length, opts)
  if (ips_match(r))
    print_error "Incoming read may match a known signature"
  end
  return r
end

def write(buf, opts = {})

    return 0
  end
  super(buf, opts)
