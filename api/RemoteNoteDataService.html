def update_note(opts)
  path = NOTE_API_PATH
  if opts && opts[:id]
    id = opts.delete(:id)
    path = "#{NOTE_API_PATH}/#{id}"
  end
  json_to_mdm_object(self.put_data(path, opts), NOTE_MDM_CLASS)
end
