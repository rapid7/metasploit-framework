def self.normalize_code(code, headers)
  code = code.lines.map { |line|
    if line =~ /^\s*#include <([[:print:]]+)>$/
      h = headers.include("#{$1}")
      %Q|#{h}\n|
    else
      line
    end
  }.join

  code
end
