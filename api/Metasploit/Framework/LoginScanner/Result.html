def to_h
  result_hash = credential.to_h
  result_hash.merge!(
      access_level: access_level,
      address: host,
      last_attempted_at: DateTime.now,
      origin_type: :service,
      port: port,
      proof: proof,
      protocol: protocol,
      service_name: service_name,
      status: status
  )
  result_hash.delete_if { |k,v| v.nil? }
end
