def update_service(opts)
  path = SERVICE_API_PATH
  if opts && opts[:id]
    id = opts.delete(:id)
    path = "#{SERVICE_API_PATH}/#{id}"
  end
  json_to_mdm_object(self.put_data(path, opts), SERVICE_MDM_CLASS)
end
